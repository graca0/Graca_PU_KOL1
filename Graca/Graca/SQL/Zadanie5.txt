CREATE PROCEDURE AddStudent
    @imie NVARCHAR(255),
    @nazwisko NVARCHAR(255),
    @grupaId INT = NULL
AS
BEGIN
    INSERT INTO Studenci (Imie, Nazwisko, IDGrupy)
    VALUES (@imie, @nazwisko, @grupaId)
END
